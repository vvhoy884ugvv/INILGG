-- وظيفة إغلاق وفتح القائمة
local isOpen = true
local function ToggleMenu()

end

-- تأثير ألوان RGB مع الحفاظ على الصورة
local function RainbowEffect()
    while true do
        for i = 0, 1, 0.01 do
            local color = Color3.fromHSV(i, 1, 1) -- تحويل إلى ألوان الطيف المتغيرة
            ToggleButton.ImageColor3 = color -- تغيير لون الصورة فقط وليس الزر بالكامل
            task.wait(0.05) -- سرعة تغيير الألوان
        end
    end
end
task.spawn(RainbowEffect) -- تشغيل تأثير الألوان
