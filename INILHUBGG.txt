local CoreGui = game:GetService("CoreGui")
local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")

local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "INIL HUB" -- تغيير اسم السكربت
ScreenGui.Parent = CoreGui

local windowsize = UDim2.new(0.7, 0, 0.8, 0) -- تكبير القائمة لتناسب الجوال

-- إنشاء نافذة
local Window = Instance.new("Frame")
Window.Name = "Window"
Window.BackgroundColor3 = Color3.fromRGB(39, 39, 39)
Window.Position = UDim2.new(0.5, 0, 0.5, 0)
Window.AnchorPoint = Vector2.new(0.5, 0.5)
Window.Size = windowsize -- تطبيق الحجم الجديد
Window.Transparency = 0.5
Window.BorderSizePixel = 0
Window.Draggable = false
Window.ClipsDescendants = false
Window.SizeConstraint = "RelativeYY"
Window.Parent = ScreenGui

-- زر خارج القائمة
local ToggleButton = Instance.new("ImageButton")
ToggleButton.Visible = true
ToggleButton.Name = "ToggleButton"
ToggleButton.BorderSizePixel = 0
ToggleButton.Image = "rbxassetid://122739889648244" -- إضافة الصورة المطلوبة
ToggleButton.ImageTransparency = 0
ToggleButton.Active = true
ToggleButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0) -- يبدأ باللون الأحمر
ToggleButton.Position = UDim2.new(0.05, 0, 0.05, 0) -- يبقى خارج القائمة
ToggleButton.Size = UDim2.new(0.1, 0, 0.1, 0)
ToggleButton.Transparency = 0
ToggleButton.BackgroundTransparency = 0
ToggleButton.BorderSizePixel = 0
ToggleButton.ZIndex = 5
ToggleButton.SizeConstraint = "RelativeXY"
ToggleButton.Parent = ScreenGui -- يكون خارج القائمة وليس داخل `Window`

-- وظيفة إغلاق وفتح القائمة
local isOpen = true
local function ToggleMenu()
    isOpen = not isOpen
    Window.Visible = isOpen
end
ToggleButton.MouseButton1Click:Connect(ToggleMenu)

-- تأثير ألوان الـ RGB (Rainbow Effect)
local function RainbowEffect()
    while true do
        for i = 0, 1, 0.01 do
            local color = Color3.fromHSV(i, 1, 1) -- تحويل إلى ألوان الطيف المتغيرة
            ToggleButton.BackgroundColor3 = color
            task.wait(0.05) -- سرعة تغيير الألوان
        end
    end
end
task.spawn(RainbowEffect) -- تشغيل تأثير الألوان

return ScreenGui, Window, ToggleButton
